SELECT ShelterCode, AnimalName,
NeuteredDate, AnimalComments
FROM animal
WHERE Neutered = 1 AND 
NeuteredDate BETWEEN '$ASK DATE Enter from date$' AND '$ASK DATE Enter to date$'
ORDER BY NeuteredDate
