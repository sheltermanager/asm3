Shelter animals who do not have a description
###

###
SELECT a.AnimalName, a.ShelterCode, a.AnimalAge, a.DisplayLocation
FROM animal a
WHERE a.Archived = 0 AND (a.AnimalComments Is Null OR a.AnimalComments = '') 
ORDER BY a.AnimalName
