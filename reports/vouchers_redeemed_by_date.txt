Vouchers Redeemed by Date
###
Financial
###
34304/Any
###
Vouchers redeemed between two dates.
###
en
###
SELECT o.OwnerName, v.VoucherName, 
o.OwnerAddress, o.OwnerTown, o.OwnerCounty, o.OwnerPostcode, o.HomeTelephone, o.MobileTelephone,
ov.ID AS VoucherNumber, ov.VoucherCode, 
ov.DateIssued, ov.DatePresented, ov.DateExpired, ov.Value AS Amount,
ov.Comments
FROM owner o
INNER JOIN ownervoucher ov ON ov.OwnerID = o.ID
INNER JOIN voucher v ON ov.VoucherID = v.ID
WHERE ov.DatePresented >= '$ASK DATE Redeemed between$'
AND ov.DatePresented <= '$ASK DATE and$'
ORDER BY v.VoucherName, ov.DatePresented
